        \global\colveccount#1
        \begin{pmatrix}
        \colvecnext
}
\def\colvecnext#1{
        #1
        \global\advance\colveccount-1
        \ifnum\colveccount>0
                \\
                \expandafter\colvecnext
        \else
                \end{pmatrix}
        \fi

        \ifnum\rowveccount>0
                &
                \expandafter\rowvecnext

                \end{pmatrix}
        \fi
